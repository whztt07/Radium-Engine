297 // Custom assert, warn and error macros.
298 // Standard assert has two main drawbacks : on some OSes it aborts the program,
299 // and the debugger will break in assert.c or and not where the calling code
300 // uses assert(). CORE_ASSERT guarantees a breakpoint when in a debugger,
301 // and always prints a useful message.
302 // CORE_WARN_IF has the same effect but it will only print a message.
303 #ifdef CORE_ENABLE_ASSERT
304 #define CORE_ASSERT( EXP, DESC ) \
305  MACRO_START \
306  if (UNLIKELY(!(EXP))) { \
307  REPORT_FAIL(EXP, DESC, "%s:%i: Assertion `%s` failed : %s\n");\
308  BREAKPOINT(0); \
309  } else {} \
310  MACRO_END
311 
312 #define CORE_WARN_IF( EXP, DESC ) \
313  MACRO_START \
314  if (UNLIKELY((EXP))) { \
315  REPORT_FAIL(EXP, DESC, "%s:%i: WARNING `%s` : %s\n");\
316  } else{} \
317  MACRO_END
318 #else
319 #define CORE_ASSERT( EXP, DESC ) // nothing
320 #define CORE_WARN_IF( EXP, DESC ) // nothing
321 #endif
